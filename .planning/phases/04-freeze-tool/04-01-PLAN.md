    ```python
    def _git_dirty(repo_root: Path = _REPO_ROOT) -> bool:
        """True if git working tree has uncommitted changes (Pitfall 4 of 04-RESEARCH.md)."""
        try:
            out = subprocess.check_output(
                ["git", "status", "--porcelain"], cwd=repo_root, text=True
            )
            return bool(out.strip())
        except Exception:  # noqa: BLE001
            return False


    def _iso_z(mtime: float) -> str:
        """Convert POSIX mtime to ISO 8601 Z (matches _ts() format)."""
        return datetime.fromtimestamp(mtime, tz=timezone.utc).strftime("%Y-%m-%dT%H:%M:%SZ")


    def _rel(p: str | Path, root: Path) -> str:
        """Return path relative to `root`; absolute fallback if outside (Pitfall 7)."""
        try:
            return str(Path(p).resolve().relative_to(root.resolve()))
        except ValueError:
            return str(p)


    def _library_versions() -> dict[str, str]:
        """Read installed pkg versions. Critical libs missing => RuntimeError (decision Q2)."""
        out: dict[str, str] = {}
        missing_critical: list[str] = []
        for pkg in CRITICAL_LIBS:
            try:
                out[pkg] = version(pkg)
            except PackageNotFoundError:
                out[pkg] = "not-installed"
                missing_critical.append(pkg)
        for pkg in BONUS_LIBS:
            try:

            except PackageNotFoundError:
                out[pkg] = "not-installed"
        if missing_critical:
            raise RuntimeError(
                f"Critical libraries not installed: {missing_critical}. "
                f"Run `uv sync --all-extras` to install all v1.0 deps."
            )
        return out
    ```

    **Public freeze() function (~28 LOC):**
    ```python
    def freeze(
        version: str,
        force: bool = False,
        results_dir: Path | None = None,
        source: Path | None = None,
    ) -> Path:
        """Copy comparison.md to frozen/ and write sidecar manifest. Return frozen md path.

        Raises FileExistsError, FileNotFoundError, RuntimeError. CLI maps each to exit 2.
        Phase 5 in-process call: `freeze(version="1.0", force=False)`.
        """
        results_dir = results_dir or _REPO_ROOT / "evaluation" / "results"
        source = source or results_dir / "comparison.md"
        out_md = results_dir / "frozen" / f"eval-numbers-v{version}.md"
        out_manifest = out_md.with_suffix(".manifest.json")

        if not source.exists():
            raise FileNotFoundError(f"Source markdown missing: {source} (run compare.py first)")

        if out_md.exists() and not force:
            raise FileExistsError(
                f"{out_md} already frozen — bump version or pass --force"
            )

        lib_versions = _library_versions()  # raises RuntimeError if any critical lib missing

        # Per-tier provenance via aggregate_tier (DO NOT re-glob)
        per_tier: dict[str, dict] = {}
        for t in SUPPORTED_TIERS:
            row = aggregate_tier(t, results_dir)
            if row is None:
                per_tier[f"tier-{t}"] = {"status": "missing"}
                continue
            entry: dict = {
                "status": "present",
                "generation_model": row.get("model"),
                "capture_timestamp": row.get("timestamp"),
                "capture_git_sha": row.get("git_sha"),
                "queries_path": _rel(row["queries_path"], results_dir),
                "queries_mtime": _iso_z(Path(row["queries_path"]).stat().st_mtime),
            }
            if row.get("cost_path"):
                entry["cost_path"] = _rel(row["cost_path"], results_dir)
                entry["cost_mtime"] = _iso_z(Path(row["cost_path"]).stat().st_mtime)
            if row.get("metrics_path"):
                entry["metrics_path"] = _rel(row["metrics_path"], results_dir)
                entry["metrics_mtime"] = _iso_z(Path(row["metrics_path"]).stat().st_mtime)
            per_tier[f"tier-{t}"] = entry

        judge_model, judge_emb = _detect_judge_provenance(results_dir)

        manifest = {
            "$schema_version": "1.0",
            "version": version,
            "frozen_at": _ts(),
            "git_sha": _git_sha(),
            "git_dirty": _git_dirty(),
            "source_markdown": _rel(source, results_dir),
            "source_markdown_mtime": _iso_z(source.stat().st_mtime),
    ```
